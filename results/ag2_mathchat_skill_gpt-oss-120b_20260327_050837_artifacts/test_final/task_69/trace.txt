from fractions import Fraction
from math import gcd
from itertools import product, combinations_with_replacement

def integer_sum_and_reciprocal_sum(tup):
    """Return True iff sum(tup) and sum(1/t) are both integers."""
    s = sum(tup, Fraction(0, 1))
    r = sum(Fraction(t.denominator, t.numerator) for t in tup)  # 1/t
    return s.denominator == 1 and r.denominator == 1

def generate_rationals(bound):
    """Yield all positive reduced fractions a/b with 1 ≤ a,b ≤ bound."""
    for a in range(1, bound + 1):
        for b in range(1, bound + 1):
            if gcd(a, b) == 1:          # reduced
                yield Fraction(a, b)

def search(n, bound):
    """Return a set of n‑tuples (as tuples of Fractions) satisfying the condition."""
    rats = list(generate_rationals(bound))

    for tup in product(rats, repeat=n):
        if integer_sum_and_reciprocal_sum(tup):
            solutions.add(tup)
    return solutions
