A chord meets the vertical diameter exactly when its endpoints lie on opposite

\[

import random
import math

def random_chord():
    """Return a chord as a pair of angles (a,b) with a,b in different quadrants."""
    while True:
        a = random.random() * 2*math.pi
        b = random.random() * 2*math.pi
        qa, qb = int(a // (math.pi/2)), int(b // (math.pi/2))
        if qa != qb:                     # endpoints must be in different quadrants
            return (a, b)

def chords_intersect(ch1, ch2):
    """Return True if the two chords intersect inside the disk."""
    a, b = ch1
    c, d = ch2
    # Helper: orientation test using cross products
    def orient(p, q, r):
        # p,q,r are angles; convert to points on unit circle
        px, py = math.cos(p), math.sin(p)
        qx, qy = math.cos(q), math.sin(q)
        rx, ry = math.cos(r), math.sin(r)
        return (qx-px)*(ry-py) - (qy-py)*(rx-px)

    # chords (a,b) and (c,d) intersect iff each segment straddles the line of the other
    return (orient(a, b, c) * orient(a, b, d) < 0) and \
           (orient(c, d, a) * orient(c, d, b) < 0)

def prob_intersection(trials=200000):
    cnt = 0
    for _ in range(trials):
        ch1 = random_chord()
        ch2 = random_chord()
        if chords_intersect(ch1, ch2):
            cnt += 1
    return cnt / trials

print(prob_intersection())
