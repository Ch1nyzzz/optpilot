import itertools

def steps_until_all_tails(config):
    """
    Simulate Harry's process.
    config is a list of 0 (T) and 1 (H).
    Returns the number of steps needed to reach all 0's.
    """
    steps = 0
    while True:
        k = sum(config)                 # number of heads
        if k == 0:                       # all tails → stop
            break
        # flip the k‑th coin (1‑based indexing)
        idx = k - 1                      # convert to 0‑based index
        config[idx] ^= 1                # flip 0↔1
        steps += 1
    return steps

def brute_force_average(n):
