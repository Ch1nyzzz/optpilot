\[

\[
J=(x_{J},-r_{a})\qquad (x_{J}>c)

C_{1}=(x_{J},0).

\[

A_{1}&= \text{proj}_{BC}(J),\\[2mm]

C_{1}&= \text{proj}_{AB}(J)=(x_{J},0).
\end{aligned}
\]
