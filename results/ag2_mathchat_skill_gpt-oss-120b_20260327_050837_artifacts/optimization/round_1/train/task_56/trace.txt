\[
n = \prod_{i} p_i^{a_i},
\]

# A number n is powerful if for every prime factor p of n, p^2 divides n.

import sympy as sp

options = {
    'A': 392,
    'B': 336,
    'C': 300,
    'D': 297
}

def is_powerful(n):
    """Return True if n is powerful, else False."""
    # Get the prime factorization as a dictionary {prime: exponent}
    fac = sp.factorint(n)
    # For each prime factor, its exponent must be at least 2
    return all(exp >= 2 for exp in fac.values())

# Evaluate each option
results = {}
for label, num in options.items():
    results[label] = (num, sp.factorint(num), is_powerful(num))

# Print the results in a readable way
for label, (num, fac, powerful) in results.items():
    print(f"Option {label}: {num}")
    print(f"  Prime factorization: {fac}")
    print(f"  Powerful? {'Yes' if powerful else 'No'}\n")

# Determine which option(s) are powerful
powerful_options = [label for label, (_, _, pw) in results.items() if pw]
print("Powerful option(s):", powerful_options)
```
