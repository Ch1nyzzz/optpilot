def count_colorings():
    # edges: label them
    # squares positions: (0,0) bottom-left, (1,0) bottom-right, (0,1) top-left, (1,1) top-right
    # edges: each unit segment identified by its coordinates and orientation
    # We'll index edges 0..11
    edges = []
    # horizontal edges: y from 0 to 2, x from 0 to1 (since unit length)
    for y in range(3):
        for x in range(2):
            edges.append(('h', x, y))
    # vertical edges: x from 0 to2, y from 0 to1
    for x in range(3):
        for y in range(2):
            edges.append(('v', x, y))
    # verify length 12
    assert len(edges)==12
    # map each square to its 4 edge indices
    square_edges = {}
    # squares coordinates (x,y) lower-left corner
    for sx in range(2):
        for sy in range(2):
            # bottom edge horizontal at (sx, sy)
            bottom = edges.index(('h', sx, sy))
            top = edges.index(('h', sx, sy+1))
            left = edges.index(('v', sx, sy))
            right = edges.index(('v', sx+1, sy))
            square_edges[(sx,sy)] = [bottom, right, top, left]  # order not important
    count=0
    for coloring in itertools.product([0,1], repeat=12): # 0 red,1 blue maybe
        ok=True
        for sq, idxs in square_edges.items():
            reds = sum(1 for i in idxs if coloring[i]==0)
            blues = sum(1 for i in idxs if coloring[i]==1)
            if reds!=2 or blues!=2:
                ok=False
                break
        if ok:
            count+=1
    return count
