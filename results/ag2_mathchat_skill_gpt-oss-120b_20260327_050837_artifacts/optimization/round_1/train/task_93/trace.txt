# ------------------------------------------------------------
#  Find the minimal value of  sum_{i=1}^n floor(a_i / i)
#  where (a_1,…,a_n) is a permutation of (1,…,n).
# ------------------------------------------------------------
import itertools
from math import floor

def min_sum(n):
    """return minimal possible sum for a given n (brute force)"""
    best = None
    best_perm = None
    for perm in itertools.permutations(range(1, n+1)):
        s = sum(floor(perm[i] / (i+1)) for i in range(n))
        if best is None or s < best:
            best = s
            best_perm = perm
    return best, best_perm
